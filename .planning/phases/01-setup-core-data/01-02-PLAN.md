private async Task<DateTime> PullAndSave(HypixelApi api, int i, DateTime lastUpdate)
{
    var tryCount = 0;
    var maxRetries = 100;
    while (tryCount < maxRetries)
    {
        var result = await api.GetBazaarProductsAsync();
        
        // Check if timestamp changed - KEY BEHAVIOR
        if (result.LastUpdated <= lastUpdate)
        {
            tryCount++;
            if (tryCount % 10 == 1)
                Console.WriteLine($" - Bazaar not updated after {tryCount} attempts...");
            await Task.Delay(500);  // Wait 500ms between checks
            continue;
        }
        
        // Process and save data...
        return result.LastUpdated;
    }
    return lastUpdate;  // Return unchanged if max retries hit
}
